
# =======================================================
# KONFIGURASI NGINX FINAL - UNTUK VM TANPA BUILD STEP
# =======================================================

# File: /etc/nginx/sites-available/default

server {
    listen 80;
    server_name _;

    root /home/clusterkobapwk/warung-app/sim-rw;
    index index.html;

    # 1. Tambahkan MIME type eksplisit untuk file TypeScript/React
    include /etc/nginx/mime.types;
    types {
        application/javascript tsx;
        application/javascript ts;
    }

    location / {
        # Hanya gunakan index.html sebagai fallback untuk rute non-file
        try_files $uri $uri/ /index.html;
    }

    # 2. Blokir Nginx agar tidak memberikan index.html untuk file source code yang hilang
    # Ini akan mencegah error "expected expression, got '<'"
    location ~* \.(tsx|ts|js|jsx|css|png|jpg|jpeg|gif|ico|svg)$ {
        try_files $uri =404;
        add_header Content-Type application/javascript;
    }

    # 3. Proxy API ke Backend Node.js
    location /api/ {
        proxy_pass http://localhost:5000/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}

# JANGAN LUPA: sudo systemctl restart nginx
